#1. The URL structure for the site

# To list all topics
http://localhost:8080/api/topics/

# To list all comments for a topic
http://localhost:8080/api/topics/1/comments

# To list all comments 
http://localhost:8080/api/comments/

#To list individual topics
http://localhost:8080/api/topics/1


# get all topics
http://localhost:8080/api/get/topics

# get all comments
http://localhost:8080/api/get/comments

# post topics
http://localhost:8080/api/post/topics

# post comments
http://localhost:8080/api/post/comments

#get individual topics
http://localhost:8080/api/get/topics/1

# get individual comments
http://localhost:8080/api/get/comments/1

# post individual topics
http://localhost:8080/api/post/topics/1

# post individual comments
http://localhost:8080/api/post/comments/1

# get specific comments for a topic
http://localhost:8080/api/get/topics/1/comments

# post specific comments for a topic
http://localhost:8080/api/post/topics/1/comments
